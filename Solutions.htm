  • for( int i = 1 ; i<=n ; i++ ){
  • printf("Enter no. %d",i);
  • scanf("%d",&t[i]);
  • }
  • long int pr = 1;
  • // Now a loop for multiplication
  • for( int ione = 1 ; ione<=n ; ione++ ){
  • pr = pr * t[i];
  • }
